using UnityEngine;

public class PlayerController : MonoBehaviour
{
    public float hareketHizi = 5f;
    public float ziplamaGucu = 10f;

    private Rigidbody2D rb;
    private bool yerdeMi = true;

    void Start()
    {
        rb = GetComponent<Rigidbody2D>();
    }

    void Update()
    {
        float yatay = Input.GetAxis("Horizontal");
        rb.velocity = new Vector2(hareketHizi * yatay, rb.velocity.y);

        if (Input.GetKeyDown(KeyCode.Space) && yerdeMi)
        {
            rb.AddForce(Vector2.up * ziplamaGucu, ForceMode2D.Impulse);
            yerdeMi = false;
        }
    }

    void OnCollisionEnter2D(Collision2D temas)
    {
        if (temas.gameObject.CompareTag("Ground"))
        {
            yerdeMi = true;
        }

        if (temas.gameObject.CompareTag("Obstacle"))
        {
            Debug.Log("Engelle çarpışma oldu!");
        }
    }
}
